        A [11001] & B [00111] & C [00101] & D [00110] & E [10100] & F [01000] & G [00011] \\
        H [11110] & I [10001] & J [01101] & K [10101] & L [11100] & M [01100] & N [11111]
    \end{tabular}
\end{table}
